on_siege_lost_province =  {
	if = {
		limit = { 
			ROOT = { has_province_modifier = effelai_hub_attuned } 
			FROM = { NOT = { has_country_flag = effelai_attuned_path } }
			FROM = { NOT = { tag = REB} }
		}
		loop_effelai_hub_lost_effect = yes
	}
}

		limit = {
			unit_owner = {
